using UnityEngine;

public class YourClassName : MonoBehaviour
{
    public GameObject[] m_Items;
    public Transform Content;
    public bool SetLoveLock;
    public List<LoveData> LoveList;
    public List<Sprite> LoveSprites;
    public int showLoveNum;

    void Start()
    {
        switch (type)
        {
            case ButtonType.LovePanel:
                print("进入情缘");
                HourseSelect.gameObject.SetActive(false);
                FriendSelect.gameObject.SetActive(false);
                LoveSelect.gameObject.SetActive(true);

                //PanelDic[go.name].SetActive(true);
                imgTop.GetComponent<Image>().sprite = myImageSprite[0];

                #region 进入时候生成
                deleteContent(OnContentDeleted);
                #endregion
                break;
        }
    }

    private void OnContentDeleted()
    {
        for (int i = 0; i < showLoveNum; i++)
        {
            GameObject Go = Instantiate(m_Items[0]);
            Go.name = LoveSprites[i].name;
            Go.transform.localScale = Vector3.one;
            Go.transform.SetParent(Content);
            Go.GetComponent<LoveItem>().SetLoveItemUIData(SetLoveLock, i, LoveList[i], LoveSprites[i]);
            Go.SetActive(false);
        }


        InstanceNullObject(LockBeatuty, LoveSprites[showLoveNum]);

        print("showLoveNum是" + showLoveNum);
        for (int i = 0; i < showLoveNum; i++)
        {
            Content.GetChild(i).gameObject.SetActive(true);
            print(" Content.GetChild(i).gameObject是" + Content.GetChild(i).gameObject.name);
        }
    }

    private void deleteContent(System.Action callback)
    {
        // 获取当前 Content 的所有子节点
        Transform[] childTransforms = Content.GetComponentsInChildren<Transform>(true);
        // 遍历所有子节点并删除
        foreach (Transform child in childTransforms)
        {
            // 只销毁子节点，而不是父节点本身
            if (child!= Content)
            {
                Destroy(child.gameObject);
            }
        }
        callback?.Invoke();
    }
}
