### 委托
```lua
Lesson7 = CS.Lesson7
local obj = Lesson7()

-- 用C#中的委托来装lua中的函数
local fun = function()
	print("Lua函数Fun")
end


-- Lua中没有复合运算符 不能+=
-- 如果第一次往委托中加函数是因为nil，不能直接+，第一次直接=
print("****************加函数****************")
obj.del = fun
obj.del = obj.del + fun
obj.del = obj.del + function()-- 不建议这样写,不好减
	print("临时申明的函数")
end
-- 委托执行
obj.del()
print("****************减函数****************")
obj.del = obj.del - fun
obj.del = obj.del - fun
obj.del()

-- 清空所有存储的函数
obj.del = nil
```
