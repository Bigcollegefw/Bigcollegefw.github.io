```lua
Lesson4 = CS.Lesson4
-- 使用静态方法
-- CS.命名空间.类名.静态方法名()
Lesson4.Eat()

-- 成员方法，实例化出来后使用
local obj = Lesson4()
obj:Speak("哈哈哈哈哈哈哈")

-- 拓展方法,和使用成员方法一致
obj:Move() -- 要加上[LuaCallCSharp]
```
