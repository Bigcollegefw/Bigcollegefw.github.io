### 数组
```lua
local obj = CS.Lesson3()

--Lua使用C#数组相关知识
--如何获取长度 因为会保留C#里面的结构 元语言 userdata
-- C#怎么用，lua就怎么用，不能使用#取获取长度
print(obj.array.Length)

-- 访问元素
print(obj.array[0])

-- 遍历，数组是C#的规则，按C#的来
-- Lua中是能取到最大值所以减一
for i = 0,obj.array.Length - 1 do
	print(obj.array[i])
end

-- Lua中创建C#的数组
```
